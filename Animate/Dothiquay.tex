 \usepackage[utf8]{vietnam}

\begin{document}
\begin{animateinline}[controls,autoplay,palindrome,loop]{8}
\multiframe{30}{i=0+1}{
\begin{tikzpicture}[>=stealth]
\def\hsf(#1){0.5*(-(#1)^3+3*(#1)^2+(#1)-3)}
\draw [->] (-5,0,0) -- (5,0,0) node [at end, right] {$x$};
\draw [->] (0,-3,0) -- (0,5,0) node [at end, left] {$y$};
\draw [->] (0,0,-4) -- (0,0,4) node [at end, left] {$z$};
\foreach \k in {0,...,\i}{
\pgfmathsetmacro{\l}{\k *12}
\draw[violet,rotate around x=\l] plot[domain=1.25:2.75,smooth,samples=100](\x,{\hsf(\x)});}
\end{tikzpicture}
}
\end{animateinline}

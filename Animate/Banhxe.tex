\draw[white] (-\xmin,-\r)-- (\xmax,-\r);
\draw[white,domain=0:\i,smooth,samples=200,variable=\t] plot({\r *\t *pi/180 +\r *cos(-90-\t)},{\r * sin(-90-\t)});
\begin{scope}[shift={(\xtam,0)},rotate=-\i ]
\draw[white] (0,0) circle (\r);
\fill[ball color=red] (0,-\r) circle (0.2);
\end{scope}
\end{tikzpicture}

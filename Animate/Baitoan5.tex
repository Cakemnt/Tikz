\def\r{3}\def\rm{1}
\pgfmathsetmacro{\b}{0.5*(\r+\rm)}
\pgfmathsetmacro{\a}{sqrt(\r *\rm)}
\pgfmathsetmacro{\c}{0.5*(\r-\rm)}
\begin{animateinline}[controls,autoplay,loop]{8}
\multiframe{360}{i=90+1}{
%\foreach \i in {90,...,449}{
\begin{tikzpicture}[thick,line join=round]
\pgfmathsetmacro{\rh}{(\r *(\r-\rm)*(1-cos(90-\i)))/(\r+\rm-(\r-\rm)*cos(90-\i))}
\path 
(0:0) coordinate (O)
(90:\r) coordinate (A)
(90:\r-\rm) coordinate (O1)
(\i:\r-\rh) coordinate (O2)
;
\clip (O) circle (1.2*\r);
\draw (O)--(A) (O)--(O2)--(O1);
\draw[blue] (90:\c) ellipse ({\a} and {\b});
\draw 
(O) circle (\r)
(O1) circle (\rm)
(O2) circle (\rh)
;
\foreach \x in {A,O,O1,O2}{\fill[orange] (\x) circle (1pt);}
\end{tikzpicture}
}
\end{animateinline}

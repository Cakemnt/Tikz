	;
	\draw[fill=blue!30] (A)--++(\g:20*\r)node[midway,sloped,above]{10cm}--++(0:18*\r)--++(\g-180:20*\r)--cycle;
	\draw[fill=green!30](B)--++(\g:20*\r)--++(90:10*\r)--++(\g-180:20*\r)--cycle;

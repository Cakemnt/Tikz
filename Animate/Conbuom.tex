\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{calc,intersections}
\usepackage{animate}
\begin{document}
%==================
\pagecolor{black}
\def\r{1}
\def\a{2}
\def\b{1.25}
\begin{animateinline}[controls,autoplay,palindrome,loop]{16}
\multiframe{90}{i=90+1}{
%\foreach \i in {90,...,179,180,...,90}{
\begin{tikzpicture}[green]
\def\go{\i}
\pgfmathsetmacro\xo{-\a *\r *cos(\go)}
\pgfmathsetmacro\yo{-\b *\r *sin(\go)}
\clip (-2*\a *\r,-2*\a *\r) rectangle (2*\a *\r,2*\a *\r);
\draw 
(180:2*\a *\r)--(0:2*\a *\r)
(-90:2*\a *\r)--(90:2*\a *\r)
;
\draw[name path=li] (-135:2*\a *\r)--(45:2*\a *\r);
\draw[name path=el,fill=blue!50,opacity=.5] (\xo,\yo) ellipse ({\a *\r} and {\b *\r});
\path[name intersections={of=el and li}] (intersection-2) coordinate (P);
\draw[fill=blue!50,opacity=.5] (\yo,\xo) ellipse ({\b *\r} and {\a *\r});
\draw[fill=orange!50,opacity=.5]
($(P)+(-90:\r)$) circle (\r)
($(P)+(180:\r)$) circle (\r)
;
\end{tikzpicture}
}
\end{animateinline}
